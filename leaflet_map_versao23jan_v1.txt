<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script src="https://unpkg.com/leaflet-draw@1.0.4/dist/leaflet.draw.js"></script>
<script>
  const googleSat = L.tileLayer('http://{s}.google.com/vt/lyrs=s&x={x}&y={y}&z={z}', {
    maxZoom: 20, subdomains:['mt0','mt1','mt2','mt3'], attribution: '© Google Maps'
  });
  const map = L.map('map', { center: [37.2555, -8.3965], zoom: 10, layers: [googleSat] });

  const drawnItems = new L.FeatureGroup().addTo(map);
  let parcelsLayer = new L.FeatureGroup().addTo(map);

  // Configuração da Barra de Desenho
  const drawControl = new L.Control.Draw({
    edit: { featureGroup: drawnItems },
    draw: { polygon: true, polyline: false, rectangle: true, circle: false, marker: false, circlemarker: false }
  });
  map.addControl(drawControl);

  function getOwnerColor(info) { 
    return info === 'Sim' ? '#2e7d32' : '#c62828'; 
  }

  function loadParcels(geojson) {
    console.log('Loading parcels into map:', geojson);
    
    // Limpa desenhos temporários feitos pelo utilizador
    drawnItems.clearLayers();
    if (parcelsLayer) { map.removeLayer(parcelsLayer); }
    parcelsLayer = L.geoJSON(geojson, {
        style: (f) => ({
            color: getOwnerColor(f.properties.tem_proprietario),
            weight: 3, fillOpacity: 0.4, fillColor: getOwnerColor(f.properties.tem_proprietario)
        }),
        onEachFeature: (f, l) => {
            l.bindPopup(`<b>${f.properties.name}</b><br>Proprietário: ${f.properties.tem_proprietario === 'yes' ? 'Sim' : 'Não'}`);
        }
    }).addTo(map);
    if (geojson.features.length > 0) map.fitBounds(parcelsLayer.getBounds());
  }

  // Evento de criação de desenho
  map.on('draw:created', function (e) {
    const layer = e.layer;
    drawnItems.clearLayers(); // Mantém apenas um desenho por vez
    drawnItems.addLayer(layer);
    
    // Envia a geometria para o Gradio
    window.parent.postMessage({
        type: 'parcel_geometry',
        data: layer.toGeoJSON().geometry
    }, '*');
  });

  window.addEventListener('message', (e) => {
    if (e.data.type === 'load_parcels') loadParcels(e.data.geojson);
  });

  map.whenReady(() => window.parent.postMessage({type: 'request_parcels'}, '*'));
</script>
