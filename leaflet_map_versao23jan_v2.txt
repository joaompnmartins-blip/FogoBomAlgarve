  <link rel="stylesheet" href="https://unpkg.com/leaflet-draw@1.0.4/dist/leaflet.draw.css"/>
  <style>
    body { margin: 0; padding: 0; }
    #map { height: 500px; width: 100%; }
    /* Estilo da Legenda */
    .legend { 
        background: white; 
        padding: 10px; 
        border-radius: 8px; 
        box-shadow: 0 1px 5px rgba(0,0,0,0.4); 
        line-height: 18px; 
        color: #333; 
        font-family: Arial, sans-serif; 
    }
    .legend h4 { margin: 0 0 8px 0; font-size: 14px; border-bottom: 1px solid #ddd; padding-bottom: 4px; }
    .legend-item { display: flex; align-items: center; margin-bottom: 4px; font-size: 12px; }
    .legend-color { width: 16px; height: 16px; margin-right: 8px; border: 1px solid #666; opacity: 0.8; }
  </style>

<div id="map"></div>
<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script src="https://unpkg.com/leaflet-draw@1.0.4/dist/leaflet.draw.js"></script>
<script>
  // --- Mapas Base ---
  const googleSat = L.tileLayer('http://{s}.google.com/vt/lyrs=s&x={x}&y={y}&z={z}', {
    maxZoom: 20, subdomains:['mt0','mt1','mt2','mt3'], attribution: '© Google Maps'
  });
  
  const osm = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap contributors'
  });

  const map = L.map('map', { 
    center: [37.2555, -8.3965], 
    zoom: 10, 
    layers: [googleSat] 
  });

  // Controlo de Camadas (Base Maps)
  const baseMaps = {
    "Satélite (Google)": googleSat,
    "OpenStreetMap": osm
  };
  L.control.layers(baseMaps).addTo(map);

  const drawnItems = new L.FeatureGroup().addTo(map);
  let parcelsLayer = new L.FeatureGroup().addTo(map);

  // Barra de Desenho
  const drawControl = new L.Control.Draw({
    edit: { featureGroup: drawnItems },
    draw: { polygon: true, polyline: false, rectangle: true, circle: false, marker: false, circlemarker: false }
  });
  map.addControl(drawControl);

  // Função de Cores por Proprietário
  function getOwnerColor(info) { 
    // Aceita 'yes' (do DB) ou 'Sim' (do Radio)
    return (info === 'Sim' || info === 'yes') ? '#2e7d32' : '#c62828'; 
  }

  // --- Legenda ---
  const legend = L.control({ position: 'bottomright' });
  legend.onAdd = function() {
    const div = L.DomUtil.create('div', 'legend');
    div.innerHTML = `
      <h4>Proprietário Info</h4>
      <div class="legend-item"><div class="legend-color" style="background:#2e7d32"></div> Sim / Identificado</div>
      <div class="legend-item"><div class="legend-color" style="background:#c62828"></div> Não / S. Info</div>
    `;
    return div;
  };
  legend.addTo(map);

  // Função para carregar as parcelas
  function loadParcels(geojson) {
    drawnItems.clearLayers();
    if (parcelsLayer) { map.removeLayer(parcelsLayer); }
    
    parcelsLayer = L.geoJSON(geojson, {
        style: (f) => ({
            color: getOwnerColor(f.properties.tem_proprietario),
            weight: 2, 
            fillOpacity: 0.4, 
            fillColor: getOwnerColor(f.properties.tem_proprietario)
        }),
        onEachFeature: (f, l) => {
            const props = f.properties;
            const popupContent = `
                <div style="font-family: Arial; min-width: 150px;">
                    <b style="font-size: 14px; color: #333;">${props.name || 'Sem Nome'}</b><br>
                    <hr style="margin: 5px 0;">
                    <b>🌲 Vegetação:</b> ${props.vegetation_type}<br>
                    <b>🏗️ Infraestruturação:</b> ${props.infraestrutura}<br>
                    <b>👤 Responsável Parcela:</b> ${props.responsavel || 'N/A'}<br>
                    <b>🏡 Info Proprietário:</b> ${props.tem_proprietario === 'yes' || props.tem_proprietario === 'Sim' ? 'Sim' : 'Não'}
                </div>
            `;
            l.bindPopup(popupContent);
        }
    }).addTo(map);
    
    if (geojson.features.length > 0) map.fitBounds(parcelsLayer.getBounds());
  }

  // --- Eventos ---
  map.on('draw:created', (e) => {
    const layer = e.layer;
    drawnItems.clearLayers();
    drawnItems.addLayer(layer);
    window.parent.postMessage({ type: 'parcel_geometry', data: layer.toGeoJSON().geometry }, '*');
  });

  window.addEventListener('message', (e) => {
    if (e.data.type === 'load_parcels') loadParcels(e.data.geojson);
  });

  map.whenReady(() => window.parent.postMessage({type: 'request_parcels'}, '*'));
</script>
